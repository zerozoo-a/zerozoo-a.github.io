exports.bubbleSort = function (A) {
  if (A.length < 2) return A;

  for (let i = 1; i < A.length; i++) {
    for (let j = 0; j < i; j++) {
      if (A[j] > A[i]) {
        [A[j], A[i]] = [A[i], A[j]];
      }
    }
  }

  return A;
};
