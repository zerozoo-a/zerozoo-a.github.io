  • 23 September 2023
각주:[1](배너_이미지_출처)


nest.js에서 header의 property를 바로 받아보자 #

nest.js의 컨트롤러로 넘어오는 header에서 값을 꺼내기 위해선 아래와 같이 해봅시다.

import {
  Response,
  Get,
  Headers,
} from '@nestjs/common';

  @Get('/kakaoLoginInfo')
  async kakaoLoginInfo(@Headers('authorized') authorized, @Response() res) {
    console.log('request header로 부터 꺼낸 값인 >>>',authorized);

    res.send({
      a: 'a',
    });
  }
